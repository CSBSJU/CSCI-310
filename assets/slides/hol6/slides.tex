  \begin{frame}{comparison}
  %{{{1
    \begin{center}
      \begin{tabularx}{.8\textwidth}{XX}
        \hline
        \textbf{Java} & \textbf{C}\\
        \hline
        object-oriented & procedural\\
        interpreted & compiled\\
        \texttt{String} & \texttt{char} array\\
        condition (\texttt{boolean}) & condition (\texttt{int})\\
        garbage-collected & no memory management\\
        references & pointers\\
        exceptions & error codes\\
        \hline
      \end{tabularx}
    \end{center}

    \note{
      \begin{itemize}
        \item in Java, everything is a method that is called on an object
        \item in C, everything is a function
      \end{itemize}
      \begin{itemize}
        \item in Java, source code is compiled to byte code, which is then
          interpreted by Java VM
        \item in C, source code is compiled into binary machine code
      \end{itemize}
      \begin{itemize}
        \item in Java, String is a class
        \item in C, a string is just an array of \texttt{char} values which ends
          with the \texttt{char~'\textbackslash0'}
      \end{itemize}
      \begin{itemize}
        \item in Java, the Java VM takes care of deallocating memory used
        \item in C, any memory you allocate, you must also deallocate
      \end{itemize}
    }
  %}}}1
  \end{frame}

      // file: figure2-4.c
      // Stan Warford
      // A nonsense program to illustrate global variables

      ###include## $$<stdio.h>$$

      char ch;
      int j;

      int main() {
        scanf([["]]%%%c%%[[ ]]%%%d%%[["]], &ch, &j);
        j += $$5$$;
        ch++;
        printf([["]]%%%c\n%d\n%%[["]], ch, j);
        return $$0$$;
      }
      \end{codeblock}
    \end{scriptsize}

    \begin{scriptsize}
      \begin{termblock}
      $ gcc -o figure2-4 figure2-4.c
      $ ./figure2-4
      M 419
      N
      424
      \end{termblock}
    \end{scriptsize}

    \note{
      \begin{itemize}
        \item What \emph{kind} of variables are \texttt{ch} and \texttt{j}?
        \item What would you expect for input
          '\texttt{\textasciitilde~2147483643}'?
      \end{itemize}
      \begin{itemize}
      \item This program would behave exactly the same had this program declared
        \texttt{ch} and \texttt{j} as local variables instead of global
        variables.
      \end{itemize}
    }
  %}}}1
  \end{frame}

  \begin{frame}{memory model --- part i}
  %{{{1
    \begin{description}
      \item[global variables] \hfill \\ declared outside of any function and
        remain in place throughout the execution of the entire program. they are
        stored at a fixed location in memory.
      \item[local variables] \hfill \\ declared within a function and come into
        existence when the function is called and cease to exist when the
        function terminates. they are stored on the run-time stack.
    \end{description}

    \vspace{1\baselineskip}

    \begin{figure}
      \centering
      \begin{subfigure}[b]{.30\textwidth}
        \centering
        \begin{tikzpicture}[
            x=1em,
            y=1em,
            font=\ttfamily,
            var/.style={
              draw,
              rectangle,
              anchor=south west,
              minimum width=3.0em,
              minimum height=1.5em,
            },
          ]

          % phantom shadow to align with the run-time stack
          \draw[color=bg] (-0.5,0.0) -- (3.5,0.0);
          \fill[color=bg] (-0.5,-0.5) rectangle ++(4.0,0.5);
          % global variables
          \foreach [
            count=\i,
            evaluate=\i as \y using 1.5*(\i-1)
          ] \l/\v in {j/424,ch/N}
          {
            \node[var,label=left:\l] at (0,\y) {\v};
          }
        \end{tikzpicture}
        \caption{Fixed location.}
      \end{subfigure}
      \quad
      \begin{subfigure}[b]{.30\textwidth}
        \centering
        \begin{tikzpicture}[

          % the shadow for the run-time stack
          \draw (-0.5,0.0) -- (3.5,0.0);
          \fill[pattern=north east lines] (-0.5,-0.5) rectangle ++(4.0,0.5);
          % stack variables
          \foreach [
            count=\i,
            evaluate=\i as \y using 1.5*(\i-1)

            \node[var,label=right:\l] at (0,\y) {\v};

        \end{tikzpicture}
        \caption{Run-time stack.}
